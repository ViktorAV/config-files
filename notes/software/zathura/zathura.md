set guioptions 'v'

set default-fg "#ebdbb2"
set default-bg "#282828"
set statusbar-bg "transparent"
set font "monospace normal 24"
set scroll-step 100
set scroll-page-aware 'true'

set recolor-lightcolor "#2d4356"
set recolor-keephue 'true'

map i recolor
map j scroll down
